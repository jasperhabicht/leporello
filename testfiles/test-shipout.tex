\makeatletter
\ExplSyntaxOn

% set all LaTeX dimensions to zero
\clist_map_inline:nn {
  paperwidth ,
  paperheight ,
  textwidth ,
  textheight ,
  topmargin ,
  oddsidemargin ,
  evensidemargin ,
  hoffset ,
  voffset ,
  headheight ,
  headsep ,
  footskip ,
  marginparwidth ,
  marginparsep ,
  marginparpush
} {
  \dim_set:cn {#1} { 0pt }
}

\pagestyle { empty }

% === 

\cs_new_protected:Npn \my_set_box_size:nn #1#2 {
  \dim_gset:Nn \hsize {#1}
  \dim_gset:Nn \vsize {#2}
  \dim_gset:Nn \textwidth {#1}
  \dim_gset:Nn \textheight {#2}
  \dim_gset:Nn \@colht {#2}
}

\int_new:N \g_my_flow_box_int

\cs_new_protected:Npn \my_box_break_flow:n #1 {
  \hook_gput_code:nnn { shipout / before } { break-flow } {
    % store shipout box and discard it
    \box_new:c {
      g_my_flow_ \int_use:N \g_my_flow_box_int _box
    }
    \box_gset_eq:cN {
      g_my_flow_ \int_use:N \g_my_flow_box_int _box
    } \l_shipout_box
    %\shipout_discard:

    \int_gincr:N \g_my_flow_box_int
  }

  % set dimensions for small boxes
  \int_set:Nn \g_my_flow_box_int { 1 }
  \my_set_box_size:nn { 70mm } { 70mm }
  #1

  % remove hook code after final page break
  \clearpage
  \hook_gremove_code:nn { shipout / before } { break-flow }
}

\ExplSyntaxOff
\makeatother

% create the boxes
\my_box_break_flow:n { \textbf{\thepage} \lipsum[1] \par \textbf{\thepage} \lipsum[2] \textbf{\thepage} }

% reset text body dimensions
\my_set_box_size:nn { 120mm } { 200mm }

\lipsum[2]
\par

\box_use:c { g_my_flow_ 1 _box }
\box_use:c { g_my_flow_ 2 _box }

\par

\textbf{\thepage}
